    async function analyzeIntent(intent){
        if(!intent||intent.length<10)return;
        var analysisDiv=document.getElementById('intentAnalysis');
        if(!analysisDiv){
            var panel=document.querySelector('#tabIntent .panel-content-inner');
            if(!panel)return;
            analysisDiv=document.createElement('div');analysisDiv.id='intentAnalysis';
            analysisDiv.style.cssText='margin-top:12px;padding:10px;border-radius:6px;background:var(--bg-base);border:1px solid var(--border-subtle);font-size:11px;line-height:1.6';
            panel.appendChild(analysisDiv);
        }
        analysisDiv.innerHTML='Analyzing intent...';
        var keys={create:1,build:1,make:1,add:1,implement:1,design:1,develop:1,fix:1,refactor:1,optimize:1,html:1,js:1,typescript:1,css:1,react:1,api:1,database:1,cli:1,app:1,web:1,site:1,page:1,component:1,tool:1,server:1,client:1,frontend:1,backend:1,fullstack:1,landing:1,ui:1,ux:1,button:1,form:1,table:1,modal:1,navbar:1,sidebar:1,dashboard:1,login:1,signup:1,auth:1,theme:1,storage:1,localstorage:1};
        var tokens=intent.toLowerCase().split(/[\s,\.\-]+/);
        var matches=tokens.filter(function(t){return keys[t]});
        var verbs=['create','build','make','add','implement','design','develop','fix','refactor','optimize'];
        var hasAction=tokens.some(function(t){return verbs.indexOf(t)>=0});
        var hasDomain=matches.length>0;
        var recommendations=[];
        if(hasAction&&hasDomain){
            analysisDiv.innerHTML='
"+'\u2705 GO — Intent looks viable
';
            recommendations.push('Good specificity — action + domain detected');
            recommendations.push('Proceed to launch. The architect will decompose this into tasks.');
        }else if(hasAction&&!hasDomain){
            analysisDiv.innerHTML='
"+'\u26a0 WEAK — Add more context
';
            recommendations.push('Intent has an action but no clear domain or technology');
            recommendations.push('Consider adding: what tech stack, what kind of project, what output');
        }else if(!hasAction&&!hasDomain){
            analysisDiv.innerHTML='
"+'\u274c TOO VAGUE — Add specifics
';
            recommendations.push('No clear action or domain found in intent');
            recommendations.push('Add: what to build, using what technology, what the output should be');
        }else{
            analysisDiv.innerHTML='
"+'\u26a0 REVIEW — May need refinement
';
            recommendations.push('Intent has domain keywords but unclear action');
            recommendations.push('Consider rephrasing as: Create a [what] using [tech] that [does what]');
        }
        if(intent.length<30)recommendations.push('Intent is very short — more detail helps the architect');
        if(intent.length>500)recommendations.push('Intent is very long — consider breaking into sub-projects');
        analysisDiv.innerHTML+=recommendations.map(function(r){return'
"+'\u2022 '+r+'
'}).join('');
        showToast('Intent analyzed','info');
    }
